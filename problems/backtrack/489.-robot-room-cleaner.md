class Solution:
    def cleanRoom(self, robot):
        """
        :type robot: Robot
        :rtype: None
        """
        
        def back():
            robot.turnRight()
            robot.turnRight()
            robot.move()
            robot.turnRight()
            robot.turnRight()
        
        def backtrack(cell, d):
            visited.add(cell)
            robot.clean()
            for i in range(4):
                di = (i + d) % 4
                new_cell = (cell[0] + directions[di][0], cell[1] + directions[di][1])
                if new_cell not in visited and robot.move():
                    backtrack(new_cell, di)
                    back()
                robot.turnRight()
        
        directions = [(1,0),(0,1),(-1,0),(0,-1)]
        visited = set()
        backtrack((0, 0), 0)
