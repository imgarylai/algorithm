# 394. Decode String

[394. Decode String](https://leetcode.com/problems/decode-string/)

```python
class Solution:
    def decodeString(self, s: str) -> str:
        stack = []
        digit = 0
        curr = ''
        for char in s:
            if char.isdigit():
                digit = digit * 10 + int(char)
            elif char == '[':
                stack.append((curr, digit))
                curr = ''
                digit = 0
            elif char == ']':
                prev, num = stack.pop()
                curr = prev + num * curr
            else:
                curr += char
        return curr
```

```python
class Solution:
    def decodeString(self, s: str) -> str:
        index = 0
        def helper():
            nonlocal index
            res = ''
            while index < len(s) and s[index] != ']':
                if not s[index].isdigit():
                    res += s[index]
                    index += 1
                else:
                    digit = ''
                    while index < len(s) and s[index].isdigit():
                        digit += s[index]
                        index += 1
                    digit = int(digit)
                    index += 1
                    decodedString = helper()
                    index += 1
                    while digit > 0:
                        res += decodedString
                        digit -= 1
            return res
        return helper()
```
